POLKADOT_CHAIN_ID=420420421
FLOWPAY_NETWORK_NAME="Westend Asset Hub"
FLOWPAY_PAYMENT_TOKEN_ADDRESS=0x00007a6900000000000000000000000001200000
FLOWPAY_PAYMENT_ASSET_ID=31337
